public class Solution {
    public int lastRemaining(int n) {
        if (n <= 1)
            return 1;
        int remain = n;
        int head = 1;
        int step = 1;
        boolean fromLeft = true;
        while (remain > 1) {
            //update head
            if (fromLeft)
                head += step;
            else {
                //if even number of elem, head remains, o.w. move
                if (remain % 2 != 0)
                    head += step;
            }
            //update remain, step and fromLeft
            step *= 2;
            remain /= 2;
            fromLeft = !fromLeft;
        }
        return head;
    }
}
